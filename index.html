Практика: пример кода

Алгоритм бинарного поиска на Python

def binary_search(arr, target):
    """
    Функция бинарного поиска в отсортированном массиве.
    Возвращает индекс элемента, если он найден, иначе -1.
    """
    left, right = 0, len(arr) - 1

    while left <= right:
        mid = (left + right) // 2

        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1

    return -1

# Пример использования
sorted_array = [1, 3, 5, 7, 9, 11, 13, 15]
target_number = 7
result = binary_search(sorted_array, target_number)

if result != -1:
    print(f"Элемент найден на позиции: {result}")
else:
    print("Элемент не найден в массиве")
